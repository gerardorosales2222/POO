package ejemploherencia;

/**
 *
 * @author Carmen
 */
//Clase base (SuperClase)
public class Animal {
    String nombre;
    
    public void hacerSonido(){
        System.out.println("Sonido generico de animal");
    }
}



package ejemploherencia;

/**
 *
 * @author Carmen
 */
//Clase derivada (SubClase)
public class Perro extends Animal{
    
    public void hacerSonido(){
        System.out.println("Guau!");
    }
}
